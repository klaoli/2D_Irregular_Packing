544 long double get_inner_angle(const point_t& joint, const point_t& end1, const point_t& end2) {
545  coord_t dx21 = end1.x_ - joint.x_;
546  coord_t dx31 = end2.x_ - joint.x_;
547  coord_t dy21 = end1.y_ - joint.y_;
548  coord_t dy31 = end2.y_ - joint.y_;
549  coord_t m12 = sqrt((dx21 * dx21 + dy21 * dy21));
550  coord_t m13 = sqrt((dx31 * dx31 + dy31 * dy31));

552  return 0;
553  return acos((dx21 * dx31 + dy21 * dy31) / (m12 * m13));
